lawOfCosines <- function(a, b, theta) {
    return(sqrt(a^2 + b^2 - 2 * a * b * cos(theta)))
}

Graphics

canvas <- function(x, y) {
    return(plot(x:y, type = "n", xlim = c(1, 100), ylim = c(1, 100), xlab = "", 
        ylab = "", asp = 1))
}
canvas(6, 8)
